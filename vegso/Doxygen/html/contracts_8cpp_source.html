81void Contract::invoice( const Date &today) {
82 std::ofstream outfile;
83 std::string filename = client->getName() + "_" + std::to_string(contract_ID) + ".txt";
84
85 outfile.open(filename);
86 // Write contract details
87 outfile << "Contract details:" << std::endl;
88 outfile<< "Seller details:" << std::endl;
89 outfile<<"Seller name: "<< MVM::MVM_company.getsub_Client_info() << std::endl;
90 outfile << "Client name: " << getClient()->getName() << std::endl;
91 outfile << "Client ID: " << getClient()->getsub_Client_info() << std::endl;
92 outfile << "Contract type: " << Ctype_toString() << std::endl;
93 outfile << "Contract start date: " << getCtime().getBegin().toString() << std::endl;
94 outfile << "Contract end date: " << getCtime().getEnd().toString() << std::endl;
95 outfile << "Last invoicing date: " << getLast_invoicing().toString() << std::endl;
96 outfile << "Tariff: " << std::fixed << std::setprecision(2) << getTariff() << " Ft/KWh" << std::endl;
97 outfile << "Balance: " << std::fixed << std::setprecision(2) << getBalance() << " Ft" << std::endl;
98 outfile << "Consumption since last invoicing: " << std::fixed << std::setprecision(2) << getConsumption() << " KWh" << std::endl;
99
100 // Calculate invoice details
101 double consumption = getConsumption();
102 double prev_balance =getBalance();
103 double tariff = getTariff();
104 double vat = 0.27; // Value added tax
105 double subtotal = consumption * tariff;
106 double total = subtotal * (1 + vat);
107 double new_balance = prev_balance + total;
108
109 // Write invoice details
110 outfile << std::endl;
111 outfile << "Invoice details:" << std::endl;
112 outfile << "Invoice date: " << today.toString() << std::endl;
113 outfile << "Previous balance: " << std::fixed << std::setprecision(2) << prev_balance << " Ft" << std::endl;
114 outfile << "Tariff: " << std::fixed << std::setprecision(2) << tariff << " Ft/KWh" << std::endl;
115 outfile << "Consumption: " << std::fixed << std::setprecision(2) << consumption << " KWh" << std::endl;
116 outfile << "Subtotal: " << std::fixed << std::setprecision(2) << subtotal << " Ft" << std::endl;
117 outfile << "VAT (" << std::fixed << std::setprecision(2) << vat*100 << "%): " << std::fixed << std::setprecision(2) << subtotal * vat << " Ft" << std::endl;
118 outfile << "Total: " << std::fixed << std::setprecision(2) << total << " Ft" << std::endl;
119 outfile << "New balance: " << std::fixed << std::setprecision(2) << new_balance << " Ft" << std::endl;
120
121 outfile.close();
122}

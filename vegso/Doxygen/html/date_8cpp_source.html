7std::chrono::system_clock::time_point Date::into_date(int year, int month, int day) {
8 if (year < 1900 || month < 1 || month>12 || day < 1 || day>31)
9 {
10 throw std::range_error("Hibas datum, vagy 1900 elotti");
11 }
12 std::tm tm = { 0, 0, 0, day, month - 1, year - 1900 }; // Month is 0-based, year is 1900-based
13 std::time_t tt = std::mktime(&tm);
14 return std::chrono::system_clock::from_time_t(tt);
15}
